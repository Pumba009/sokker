cmd commands:
create folder:
    mkdir folderName

delete folder: 
    rmdir /s folderName

create file:
    type nul > fileName.ext

delete file:
    del fileName.ext (bash)
    rm my_chart.ts

move file:
    move fileName.ext folder\fileName.ext

copy file:
    cp manifest.json dist/manifest.json

rename file/folder:
    ren oldName.ext newName.ext or ren "old name.txt" "new name.txt"

modify file:
    echo "hello" >> jest.config.js
    notepad jest.config.js



start project:
Skompilutj plik ts:
    npx tsc

Uruchom Vite dev server (obsluguje hot reload dla popup i content_script):
    npm run dev
