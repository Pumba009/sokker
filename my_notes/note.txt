    mkdir folderName

delete folder: 
    rmdir /s folderName

    type nul > fileName.ext

    del fileName.ext

move file:
    move fileName.ext folder\fileName.ext

    cp manifest.json dist/manifest.json

    ren oldName.ext newName.ext or ren "old name.txt" "new name.txt"
